
select IL.Name, I.SUMMARY, I.DESCRIPTION from Issues I
join ISSUESLISTS iL on i.ISSUESLIST_ID = iL.ID
order by IL.NAME;


select U.FIRSTNAME, U.LASTNAME, I.SUMMARY from USERS U
join ISSUES i on u.ID = i.USER_ID_ASSIGNEDTO
order by U.LASTNAME;


SELECT U.FIRSTNAME, U.LASTNAME, COUNT(*) AS TASKS_TOTAL
FROM USERS U JOIN ISSUES I on U.ID = I.USER_ID_ASSIGNEDTO
GROUP BY U.ID, U.LASTNAME, U.FIRSTNAME
HAVING COUNT(*) > 0
ORDER BY U.LASTNAME;
